# Lens basis order (LSB -> MSB): Square, Flower, Cloud, Fractal
LENS = ["SQUARE", "FLOWER", "CLOUD", "FRACTAL"]  # indices 0..3

def enc_mask(station: set[str]) -> int:
    """
    Input: station ⊆ {SQUARE,FLOWER,CLOUD,FRACTAL}, station ≠ ∅
    Output: m ∈ {1..15}
    """
    m = 0
    for j, name in enumerate(LENS):
        if name in station:
            m |= (1 << j)
    assert 1 <= m <= 15
    return m

def dec_mask(m: int) -> set[str]:
    """
    Input: m ∈ {1..15}
    Output: station ⊆ {SQUARE,FLOWER,CLOUD,FRACTAL}, station ≠ ∅
    """
    assert 1 <= m <= 15
    station = set()
    for j, name in enumerate(LENS):
        if (m >> j) & 1:
            station.add(name)
    return station

def enc_sigma(station: set[str]) -> int:
    """
    Output: σ ∈ Z_15 represented as {0..14}
    """
    return enc_mask(station) - 1

def dec_sigma(sigma: int) -> set[str]:
    """
    Input: σ ∈ {0..14}
    """
    assert 0 <= sigma <= 14
    return dec_mask(sigma + 1)

03.S4.c — Reference algorithms: (\mathrm{ORBIT_NEXT}) and orbit utilities
MOD = 15

def orbit_next(sigma: int) -> int:
    """
    σ ↦ σ+1 mod 15
    """
    return (sigma + 1) % MOD

def orbit_advance(sigma: int, n: int) -> int:
    """
    σ ↦ σ+n mod 15, n may be negative
    """
    return (sigma + (n % MOD)) % MOD

def orbit_next_station(station: set[str]) -> set[str]:
    """
    Transport ORBIT_NEXT to station semantics.
    """
    return dec_sigma(orbit_next(enc_sigma(station)))

def orbit_distance_forward(a: int, b: int) -> int:
    """
    Forward steps from a to reach b under ORBIT_NEXT.
    """
    return (b - a) % MOD

def orbit_distance_min(a: int, b: int) -> int:
    """
    Minimal cyclic distance on the 15-ring.
    """
    fwd = orbit_distance_forward(a, b)
    return min(fwd, MOD - fwd)

✶ Fractal — Recursive Carrier (Orbit Iteration, Route Logs, and Coarse-Graining)
03.R1 — Ontology (Objects)
03.R1.a — Orbit words and route logs
Define the orbit word monoid ((\mathbb{Z},+)) acting on (\Sigma_{15}) by (\mathrm{ORBIT_ADVANCE}(\cdot;n)).
A route log is a pair ((\sigma_0,n)) representing the map (\sigma\mapsto \sigma+n) applied at (\sigma_0), or a word (n_1n_2\cdots n_T) representing successive advances.

03.R1.b — Canonical compression of route logs
Any finite word (n_1,\dots,n_T\in\mathbb{Z}) composes to a single advance
[
n_{\mathrm{tot}}:=\sum_{t=1}^T n_t,
\qquad
\mathrm{ORBIT_ADVANCE}(\sigma_0;n_1)\circ\cdots\circ\mathrm{ORBIT_ADVANCE}(\sigma_0;n_T)
\mathrm{ORBIT_ADVANCE}(\sigma_0;n_{\mathrm{tot}}).
]
Thus a route log canonically compresses to ((\sigma_0, n_{\mathrm{tot}}\bmod 15)).

03.R1.c — Coarse observables on the station carrier
Define coarse observables on semantic stations, e.g. the weight map
[
w:\mathcal{P}^\star(\mathbb{L}4)\to{1,2,3,4},\quad w(U)=|U|.
]
Transported to (\Sigma{15}) via (\mathrm{DEC}{\sigma}), define (w(\sigma):=w(\mathrm{DEC}{\sigma}(\sigma))).

03.R1.d — Induced action on observables
For any observable (f:\Sigma_{15}\to X), define its orbit-shifted version by
[
(\mathsf{T}f)(\sigma):=f(\sigma-1\pmod{15}),
]
and iterates (\mathsf{T}^t f(\sigma)=f(\sigma-t\pmod{15})). This defines the recursion/iteration layer for station-indexed structures.
