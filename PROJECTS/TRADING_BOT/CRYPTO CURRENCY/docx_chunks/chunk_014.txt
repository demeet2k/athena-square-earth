from __future__ import annotations
from dataclasses import dataclass
from typing import List, Tuple, Literal

Digit = Literal[0, 1, 2]

def B(d: int) -> int:
    assert d >= 1
    return 15 * (3 ** (d - 1))

def W(d: int) -> int:
    assert d >= 1
    return 3 ** (d - 1)

def proj_dim(u: int, d: int, d2: int) -> int:
    """
    Projection ŽĆ_{dŌåÆd2} on indices, grouping triadic refinements.
    u is assumed canonical in [0, B(d)-1].
    """
    assert 1 <= d2 <= d
    return u // (3 ** (d - d2))

def lift_dim(u: int, d: int, t: Digit) -> int:
    """
    One-step lift ╬╗_{dŌåÆd+1}^{(t)}.
    """
    assert d >= 1
    return 3 * u + int(t)

def station(u: int, d: int) -> int:
    """
    Macro-station projection ŽĆ_st^{(d)}(u) in {0,...,14}.
    """
    return u // W(d)

def stratum(u: int, d: int) -> int:
    """
    Stratum projection ŽĆ_str^{(d)}(u) in {0,...,W(d)-1}.
    """
    return u - station(u, d) * W(d)

def digits_stratum(gamma: int, d: int) -> Tuple[Digit, ...]:
    """
    Convert gamma in [0, 3^{d-1}-1] to (t1,...,t_{d-1}) in base 3.
    Most significant digit first. For d=1, returns empty tuple.
    """
    assert d >= 1
    if d == 1:
        return tuple()
    out: List[Digit] = []
    rem = gamma
    for i in range(d - 1):
        place = 3 ** (d - 2 - i)
        ti = rem // place
        rem = rem - ti * place
        out.append(ti)  # type: ignore
    return tuple(out)  # type: ignore

def undigits_stratum(ts: Tuple[Digit, ...]) -> int:
    """
    Convert (t1,...,t_{m}) in base 3 to integer gamma.
    """
    m = len(ts)
    gamma = 0
    for i, ti in enumerate(ts):
        place = 3 ** (m - 1 - i)
        gamma += int(ti) * place
    return gamma

def gear_stack(u: int, d: int) -> Tuple[int, Tuple[Digit, ...]]:
    """
    Returns (sigma, (t1,...,t_{d-1})) for u in ╬Ż_{B_d}.
    """
    sigma = station(u, d)
    gamma = stratum(u, d)
    ts = digits_stratum(gamma, d)
    return sigma, ts

def from_gear_stack(sigma: int, ts: Tuple[Digit, ...], d: int) -> int:
    """
    Inverse reconstruction GS_d^{-1}.
    """
    assert 0 <= sigma <= 14
    assert len(ts) == max(0, d - 1)
    gamma = undigits_stratum(ts)
    assert 0 <= gamma < W(d)
    return sigma * W(d) + gamma

def multi_gear(u: int, d: int) -> Tuple[int, ...]:
    """
    Multi-gear readout MG_d(u) = (u^{(1)},...,u^{(d)}).
    """
    return tuple(proj_dim(u, d, r) for r in range(1, d + 1))
