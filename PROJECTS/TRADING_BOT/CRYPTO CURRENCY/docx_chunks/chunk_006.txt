06.S4.a — Nearest-hub computation (1D and (d)D)
from math import floor

D = [2, 3, 4, 6, 12, 36]

def dist_T1(a: float, b: float) -> float:
    """a,b in [0,1). Circular distance on T^1."""
    x = (a - b) % 1.0
    return min(x, 1.0 - x)

def hubs_1d():
    """Enumerate canonical hubs in [0,1) with denominators in D."""
    H = set()
    for den in D:
        for k in range(den):
            H.add(k / den)
    return sorted(H)

H1 = hubs_1d()

def nearest_hub_1d(phi: float) -> float:
    """Deterministic nearest hub in H1 (ties broken by smaller denominator then smaller value)."""
    phi %= 1.0
    best = None
    best_d = 1.0
    for q in H1:
        d = dist_T1(phi, q)
        if d < best_d - 1e-15:
            best, best_d = q, d
        elif abs(d - best_d) <= 1e-15:
            # tie-break: prefer smaller denominator representation; then smaller q
            # implement by comparing minimal den in D that represents q exactly
            def den_rank(x):
                for den in D:
                    if abs(x * den - round(x * den)) <= 1e-15:
                        return den
                return 10**9
            if (den_rank(q), q) < (den_rank(best), best):
                best, best_d = q, d
    return best

def snap(phi_vec):
    return [nearest_hub_1d(phi) for phi in phi_vec]

def residual(phi_vec, hub_vec):
    out = []
    for phi, h in zip(phi_vec, hub_vec):
        r = (phi - h) % 1.0
        if r >= 0.5:
            r -= 1.0
        out.append(r)
    return out

06.S4.b — Multi-axis tunnel steps and legality test
STEP = [ 1/6, -1/6, 1/12, -1/12, 1/4, -1/4, 1/3, -1/3, 1/36, -1/36, 0.0 ]

def in_step_set(x: float) -> bool:
    return any(abs((x - s) % 1.0) <= 1e-15 for s in STEP)

def legal_step(delta_vec) -> bool:
    return all(in_step_set(d) for d in delta_vec)

def apply_step(phi_vec, delta_vec):
    return [ (phi + d) % 1.0 for phi, d in zip(phi_vec, delta_vec) ]

06.S4.c — Hub-first route synthesis (coordinatewise canonical plan)
def hub_first_route(phiA, phiB):
    """
    Returns a route artifact:
      - hA, rA, hB, rB
      - hub_steps: a list of delta vectors (each coordinate in STEP)
    The hub path is constructed coordinatewise by reducing hub differences mod 1
    into a short word over {±1/36, ±1/12, ±1/6, ±1/4, ±1/3}.
    """
    hA = snap(phiA)
    hB = snap(phiB)
    rA = residual(phiA, hA)
    rB = residual(phiB, hB)

    # Route on hubs: move hA -> hB in each coordinate using 1/36 granularity
    # (since all denominators divide 36).
    hub_steps = []
    cur = hA[:]
    # deterministic axis order: 1..d
    d = len(cur)
    for i in range(d):
        # represent difference on i-th coordinate in [-1/2,1/2)
        diff = (hB[i] - cur[i]) % 1.0
        if diff >= 0.5:
            diff -= 1.0
        # exact rational step count in units of 1/36
        m = round(diff * 36)
        # apply |m| steps of ±1/36 on axis i
        step = 1/36 if m > 0 else -1/36
        for _ in range(abs(m)):
            delta = [0.0]*d
            delta[i] = step
            hub_steps.append(delta)
            cur = apply_step(cur, delta)

    return {
        "phiA": phiA, "hA": hA, "rA": rA,
        "phiB": phiB, "hB": hB, "rB": rB,
        "hub_steps": hub_steps
    }

This protocol implements the “Hub-first” schema: snap to nearest hubs, route along a hub-step generating set, then snap out.
