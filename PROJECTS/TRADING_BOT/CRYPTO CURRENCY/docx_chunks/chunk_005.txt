from dataclasses import dataclass
from typing import Callable, List, Literal, Tuple

Mode = Literal["AXIS", "GLOBAL"]

def frac(x: float) -> float:
    # Canonical fractional part in [0,1)
    return x - int(x // 1)

def q3(u01: float) -> int:
    # u01 in [0,1)
    t = int(3 * u01)
    return t if t < 3 else 2  # guard against u01=1.0 by float error

def q3_wrapped(phi: float, R: int) -> int:
    return q3(frac(R * phi))

def interleave_base4_digits(x: List[int], d: int, k: int) -> int:
    """
    x: list length d with 0 <= x[i] < 4^k
    returns interleaved J in [0, 4^(k*d)-1]
    """
    assert len(x) == d
    J = 0
    for t in range(k):
        for i in range(d):
            digit = (x[i] >> (2 * t)) & 0b11  # base-4 digit at level t
            J |= (digit << (2 * (d * t + i)))
    return J

def phi_global_from_square(x: List[int], d: int, k: int) -> float:
    J = interleave_base4_digits(x, d, k)
    return J / float(4 ** (k * d))
