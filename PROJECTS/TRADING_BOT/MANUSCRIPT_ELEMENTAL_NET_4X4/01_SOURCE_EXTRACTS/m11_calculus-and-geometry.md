x: np.ndarray,
A: np.ndarray,
B: np.ndarray,
dt: float) -> float:
"""
Spin_{dt} = e^{dt A} e^{dt B} e^{-dt A} e^{-dt B}
For MVP: use 2nd order exp approximation (stable for small dt).
Replace with expm later if needed.
"""
I = np.eye(A.shape[0], dtype=A.dtype)
def exp_approx(M: np.ndarray) -> np.ndarray:
# 2nd order: I + dt M + (dt^2/2) M^2
return I + dt * M + 0.5 * (dt ** 2) * (M @ M)
EA = exp_approx(A)
EB = exp_approx(B)
EAm = exp_approx(-A)
EBm = exp_approx(-B)
Spin = EA @ EB @ EAm @ EBm
x2 = Spin @ x
return rel_err(x, x2)

nF = F.shape[0]
nE = len(edges)
# Build d0 (edge-vertex incidence): (d0 f)(e) = f(head) - f(tail)
d0 = np.zeros((nE, nV), dtype=float)
for ei,(t,h) in enumerate(edges):
d0[ei,t] = -1.0
d0[ei,h] = +1.0
# Build d1 (face-edge incidence): (d1 α)(face) = sum_{edges in boundary} s * α(edge)
d1 = np.zeros((nF, nE), dtype=float)
for fi,(a,b,c) in enumerate(F):
face_edges = [(a,b),(b,c),(c,a)] # oriented as given by face orientation
for u,v in face_edges:
key = (min(u,v), max(u,v))
ei = edge_map[key]
# If face orientation matches global edge orientation (min->max), sign +1 else -1
t,h = edges[ei]
sgn = +1.0 if (u==t and v==h) else -1.0
d1[fi, ei] += sgn
# Boundary chain of the full domain: ∂(sum faces)
c2 = np.ones(nF, dtype=float) # all faces included with +1
boundary_edges = d1.T @ c2 # (nE,) (this is ∂2(c2)) in chain form
# Geometry-based diagonal ★ (barycentric-lumped)
# ★2: face areas
face_areas = np.zeros(nF, dtype=float)
for fi,(a,b,c) in enumerate(F):
face_areas[fi] = tri_area(V[a], V[b], V[c])
star2 = np.diag(face_areas)
# ★0: vertex dual "areas" via 1/3 of adjacent face areas
vert_area = np.zeros(nV, dtype=float)
for fi,(a,b,c) in enumerate(F):
A = face_areas[fi]
vert_area[a] += A/3.0
vert_area[b] += A/3.0
vert_area[c] += A/3.0
star0 = np.diag(vert_area)
# ★1: simplest diagonal; for deeper versions replace with true barycentric/circumcentric dual edge lengths
# Here we use edge_length as weight proxy so inner products at least scale sensibly.
edge_len = np.zeros(nE, dtype=float)
for ei,(t,h) in enumerate(edges):
edge_len[ei] = l2(V[h]-V[t])
# weight ~ dual_length / primal_length; we approximate dual_length by average adjacent face altitude ~ area/(0.5*len)
# If an edge touches k faces, dual_length ≈ sum_{faces} (area_face)/(0.5*edge_len) = 2*sum(area)/edge_len
# Then ★1_diag ≈ dual_length / edge_len = 2*sum(area)/edge_len^2
adj_area_sum = np.zeros(nE, dtype=float)
for fi in range(nF):
for ei in range(nE):
if d1[fi,ei] != 0:
adj_area_sum[ei] += face_areas[fi]
star1_diag = 2.0 * adj_area_sum / (edge_len**2 + 1e-12)
star1 = np.diag(star1_diag)
return SimplicialComplex2D(
V=V, F=F, edges=edges, edge_map=edge_map,
d0=d0, d1=d1, boundary_edges=boundary_edges,
star0=star0, star1=star1, star2=star2
)
# ----------------------------- DEC operators -----------------------------
def codifferential(d_prev: np.ndarray, star_km1: np.ndarray, star_k: np.ndarray) -> np.ndarray:
"""
δ_k : C^k -> C^{k-1}
δ_k = ★_{k-1}^{-1} d_{k-1}^T ★_k
"""
inv_star_km1 = np.diag(1.0 / (np.diag(star_km1) + 1e-12))
return inv_star_km1 @ d_prev.T @ star_k
def laplacian_0(d0: np.ndarray, star0: np.ndarray, star1: np.ndarray) -> np.ndarray:
# Δ0 = δ1 d0
δ1 = codifferential(d0, star0, star1) # C^1 -> C^0
return δ1 @ d0
def laplacian_1(d0: np.ndarray, d1: np.ndarray, star0: np.ndarray, star1: np.ndarray, star2: np.ndarray) -> np.ndarray:
# Δ1 = d0 δ1 + δ2 d1
δ1 = codifferential(d0, star0, star1) # C^1 -> C^0
δ2 = codifferential(d1, star1, star2) # C^2 -> C^1
return d0 @ δ1 + δ2 @ d1

# -------------------------- numerics / utils --------------------------
def l2(x: np.ndarray) -> float:
return float(np.linalg.norm(x.ravel()))
def rel_err(a: np.ndarray, b: np.ndarray, eps: float = 1e-12) -> float:
return l2(a - b) / (l2(a) + eps)
def sym(A: np.ndarray) -> np.ndarray:
return 0.5 * (A + A.T)
def skew(A: np.ndarray) -> np.ndarray:
return 0.5 * (A - A.T)
def stable_expm(A: np.ndarray, t: float, order: int = 8) -> np.ndarray:
"""
Pure NumPy matrix exponential via scaling + Taylor.
Good enough for small/moderate sizes and for diagnostics.
"""
n = A.shape[0]
I = np.eye(n, dtype=A.dtype)
M = t * A
# scaling
normM = np.linalg.norm(M, ord=np.inf)
s = max(0, int(np.ceil(np.log2(normM + 1e-12))))
M = M / (2 ** s)
# Taylor
X = I.copy()
term = I.copy()
for k in range(1, order + 1):
term = term @ (M / k)
X = X + term
# squaring
for _ in range(s):
X = X @ X
return X
def entropy_like(x: np.ndarray, eps: float = 1e-12) -> float:
"""
A simple entropy proxy on a state vector:
- interpret p_i = |x_i|^2 / ||x||^2
"""
p = np.abs(x)**2
p = p / (p.sum() + eps)
p = np.clip(p, eps, 1.0)
return float(-(p * np.log(p)).sum())
def condition_number(A: np.ndarray, eps: float = 1e-12) -> float:
s = np.linalg.svd(A, compute_uv=False)
smax = float(s[0])
smin = float(s[-1])
return smax / (smin + eps)
def spectral_radius(A: np.ndarray) -> float:
w = np.linalg.eigvals(A)
return float(np.max(np.abs(w)))

@dataclass
class HodgeKernel2D:
"""
Minimal 2D complex on a unit square split into two triangles.
Provides d0, d1, and a Laplacian on 0-forms and 1-forms.
"""
V: np.ndarray # (nV,2)
F: np.ndarray # (nF,3)
edges: np.ndarray # (nE,2) oriented edges tail->head
d0: np.ndarray # (nE,nV)
d1: np.ndarray # (nF,nE)
star0: np.ndarray # (nV,nV) diagonal
star1: np.ndarray # (nE,nE) diagonal
star2: np.ndarray # (nF,nF) diagonal
@staticmethod
def build() -> "HodgeKernel2D":
V = np.array([[0.,0.],[1.,0.],[1.,1.],[0.,1.]])
F = np.array([[0,1,2],
[0,2,3]], dtype=int)
# undirected edges
und = {}
for (a,b,c) in F:
for u,v in [(a,b),(b,c),(c,a)]:
key = (min(u,v), max(u,v))
und[key] = 1
keys = sorted(und.keys())
edge_map = {k:i for i,k in enumerate(keys)}
edges = np.array([[k[0], k[1]] for k in keys], dtype=int) # canonical orientation min->max
nV = V.shape[0]
nE = edges.shape[0]
nF = F.shape[0]
# d0
d0 = np.zeros((nE, nV), dtype=float)
for ei,(t,h) in enumerate(edges):
d0[ei,t] = -1.0
d0[ei,h] = +1.0
# d1
d1 = np.zeros((nF, nE), dtype=float)
for fi,(a,b,c) in enumerate(F):
face_edges = [(a,b),(b,c),(c,a)]
for u,v in face_edges:
key = (min(u,v), max(u,v))
ei = edge_map[key]
t,h = edges[ei]
sgn = +1.0 if (u==t and v==h) else -1.0
d1[fi,ei] += sgn
# geometry
def tri_area(p,q,r):
return 0.5 * abs(np.cross(q-p, r-p))
face_area = np.array([tri_area(V[a],V[b],V[c]) for (a,b,c) in F], dtype=float)
star2 = np.diag(face_area)
vert_area = np.zeros(nV, dtype=float)
for fi,(a,b,c) in enumerate(F):
A = face_area[fi]
vert_area[a] += A/3
vert_area[b] += A/3
vert_area[c] += A/3
star0 = np.diag(vert_area)
edge_len = np.array([l2(V[h]-V[t]) for (t,h) in edges], dtype=float)
# crude diagonal ★1 (upgradeable)
adj_area_sum = np.zeros(nE, dtype=float)
for fi in range(nF):
for ei in range(nE):
if d1[fi,ei] != 0:
adj_area_sum[ei] += face_area[fi]
star1_diag = 2.0 * adj_area_sum / (edge_len**2 + 1e-12)
star1 = np.diag(star1_diag)
return HodgeKernel2D(V=V,F=F,edges=edges,d0=d0,d1=d1,star0=star0,star1=star1,star2=star2)
def delta1(self) -> np.ndarray:
# δ1 = ★0^{-1} d0^T ★1
inv0 = np.diag(1.0/(np.diag(self.star0)+1e-12))
return inv0 @ self.d0.T @ self.star1
def delta2(self) -> np.ndarray:
# δ2 = ★1^{-1} d1^T ★2
inv1 = np.diag(1.0/(np.diag(self.star1)+1e-12))
return inv1 @ self.d1.T @ self.star2
def lap0(self) -> np.ndarray:
δ1 = self.delta1()
return δ1 @ self.d0
def lap1(self) -> np.ndarray:
δ1 = self.delta1()
δ2 = self.delta2()
return self.d0 @ δ1 + δ2 @ self.d1
# ----------------------------- quad-polar operators -----------------------------
@dataclass
class QuadOps:
"""
All operators act on the SAME state space dimension n (choose nV or nE).
For v1 we run on 0-forms (n=nV) to keep the demo small.
"""
D: np.ndarray # dissipative (symmetric PSD)
Omega: np.ndarray # coherent (skew-symmetric)
Sigma: np.ndarray # stochastic (Markov generator: rows sum to 0, off-diag >=0)
Psi: np.ndarray # recursion/coarse-grain projector (idempotent approx)
@staticmethod
def from_hodge_on_vertices(H: HodgeKernel2D, seed: int = 0) -> "QuadOps":
rng = np.random.default_rng(seed)
n = H.V.shape[0]
# D: Laplacian on 0-forms
D = H.lap0()
D = sym(D)
# Ω: skew transport; make a random skew operator, then corridor-lock by projecting away the constant mode
R = rng.normal(size=(n,n))
Omega = skew(R)
# kill constant direction to avoid trivial drift
ones = np.ones((n,1))
P0 = np.eye(n) - (ones @ ones.T)/(float(ones.T@ones))
Omega = P0 @ Omega @ P0
# Σ: Markov generator on vertices (graph random walk)
# Build adjacency from edges
A = np.zeros((n,n), dtype=float)
for (t,h) in H.edges:
A[t,h] = 1.0
A[h,t] = 1.0
deg = A.sum(axis=1)
Q = np.zeros((n,n), dtype=float)
for i in range(n):
for j in range(n):
if i != j and A[i,j] > 0:
Q[i,j] = A[i,j] / max(deg[i], 1.0)
Q[i,i] = -Q[i].sum()
Sigma = Q # continuous-time generator
# Ψ: coarse-grain projector (lump vertices into 2 groups: {0,2} and {1,3})
groups = np.array([0,1,0,1], dtype=int)
k = 2
Rg = np.zeros((k,n), dtype=float) # restriction (averaging)
Pg = np.zeros((n,k), dtype=float) # prolongation (replicate)
for i in range(n):
Rg[groups[i], i] = 1.0
# normalize restriction rows to average
for g in range(k):
Rg[g] /= max(Rg[g].sum(), 1.0)
for i in range(n):
Pg[i, groups[i]] = 1.0
Psi = Pg @ Rg # projector: n×n
# enforce near-idempotence numerically
Psi = sym(Psi)
return QuadOps(D=D.astype(float),
Omega=Omega.astype(float),
Sigma=Sigma.astype(float),
Psi=Psi.astype(float))
# ----------------------------- spin lab -----------------------------
@dataclass
class SpinLab:
"""
Measures commutator curvature and holonomy.
"""
@staticmethod
def comm(A: np.ndarray, B: np.ndarray) -> np.ndarray:
return A@B - B@A
@staticmethod
def spin_residual(x: np.ndarray, A: np.ndarray, B: np.ndarray, dt: float) -> float:
EA = stable_expm(A, dt, order=8)
EB = stable_expm(B, dt, order=8)
EAm = stable_expm(A, -dt, order=8)
EBm = stable_expm(B, -dt, order=8)
Spin = EA @ EB @ EAm @ EBm
return rel_err(x, Spin @ x)
@staticmethod
def bch_budget(A: np.ndarray, B: np.ndarray, dt: float) -> Dict[str,float]:
# Leading term of holonomy scales like dt^2 ||[A,B]||
C1 = SpinLab.comm(A,B)
return {
"||[A,B]||": l2(C1),
"dt^2||[A,B]||": (dt**2) * l2(C1),
"rho(A)": spectral_radius(A),
"rho(B)": spectral_radius(B),
}
# ----------------------------- No-Go detection + tunneling -----------------------------
@dataclass
class Corridor:
"""
Holds legality constraints and parameters that can be tightened by tunneling.
"""
band_mask: Optional[np.ndarray] = None # for spectral gating (length n)
eps_reg: float = 1e-10 # regularization strength
leak_gamma: float = 0.0 # metastable leakage / damping
scale_tau: float = 0.0 # scale-lift coordinate (tau)
coarse_strength: float = 1.0 # how hard Ψ is applied (0..1)
@dataclass
class NoGoReport:
kind: str
score: float
details: Dict[str, float] = field(default_factory=dict)
class NoGoDetector:
"""
Classifies structural breakage into 4 canonical types:
- KERNEL: non-injectivity / rank collapse / ill-conditioning (Square)
- ALIAS: fold overlap / out-of-band energy / false reconstruction (Flower/Air)
- FIRE: unstable amplification / expansive loop / blow-up (Fire)
- UNITARY: false unitarity / non-conservative claimed conservative (Air)
"""
@staticmethod
def detect(ops: QuadOps, corridor: Corridor, dt: float) -> NoGoReport:
D, O, S, P = ops.D, ops.Omega, ops.Sigma, ops.Psi
# KERNEL: projector ill-conditioning / near singularities
kappaP = condition_number(P + corridor.eps_reg*np.eye(P.shape[0]))
score_kernel = max(0.0, np.log10(kappaP) - 8.0) # threshold ~ 1e8
# ALIAS: out-of-band energy indicator (if mask provided)
score_alias = 0.0
if corridor.band_mask is not None:
# treat "basis" as eigenbasis of D (cheap spectral proxy)
lam, U = np.linalg.eigh(sym(D))
# low band indices from mask
mask = corridor.band_mask.astype(bool)
# if band_mask length differs, adapt to n
if mask.shape[0] == D.shape[0]:
# alias “risk” = fraction of spectrum outside chosen band (fixed)
score_alias = float((~mask).mean())
else:
score_alias = 0.0
# FIRE: blow-up / expansion: check spectral radius of exp(dt G)
# Use a proxy: dt * rho(G) too large => explosive step
G = (D + O + S + P) / 4.0
rhoG = spectral_radius(G)
score_fire = max(0.0, dt * rhoG - 0.8) # threshold ~ 0.8
# UNITARY: check whether Omega is near skew and whether D/S introduce loss
# If someone tries to run “unitary” but D or S weight is significant -> false unitarity.
unitary_defect = l2(O + O.T) # should be ~0 for skew
score_unitary = max(0.0, np.log10(unitary_defect + 1e-18) + 10.0) # if defect > 1e-10 -> positive
# Decide the dominant one
scores = {
"KERNEL": score_kernel,
"ALIAS": score_alias,
"FIRE": score_fire,
"UNITARY": score_unitary,
}
kind = max(scores, key=lambda k: scores[k])
return NoGoReport(kind=kind, score=scores[kind], details=scores)
class TunnelResolver:
"""
Minimal legal repair moves (tunneling resolutions):
- KERNEL: add regulator / renormalize (increase eps_reg)
- ALIAS: bandlimit harder (shrink band) / project to corridor
- FIRE: scale-lift tau = -log(T-t) proxy -> reduce dt / add damping
- UNITARY: enforce skew / remove dissipative parts when claiming unitary OR admit leakage
"""
@staticmethod
def apply(report: NoGoReport, corridor: Corridor) -> Corridor:
c = Corridor(**vars(corridor))
if report.kind == "KERNEL":
c.eps_reg = min(1e-2, max(c.eps_reg * 10.0, 1e-10))
elif report.kind == "ALIAS":
# tighten mask by keeping fewer modes if mask exists
if c.band_mask is not None:
mask = c.band_mask.copy()
idx = np.where(mask)[0]
if idx.size > 2:
# drop highest kept index (tighten band)
mask[idx[-1]] = False
c.band_mask = mask
elif report.kind == "FIRE":
# add leakage and scale-lift
c.leak_gamma = min(1.0, c.leak_gamma + 0.05)
c.scale_tau = c.scale_tau + 0.1
elif report.kind == "UNITARY":
# either enforce unitarity or admit irreversibility:
# choose minimal: add small leakage (decohere) so we stop pretending perfect reversibility
c.leak_gamma = min(1.0, c.leak_gamma + 0.02)
return c

@dataclass
class QuadPolarEngine:

def generator(self, alpha: np.ndarray) -> np.ndarray:
"""

def make_default_axes(n: int, seed: int = 0) -> List[AxisMap]:
"""
Produces 3 example axes to demonstrate predictability:
Axis 0: orthonormal basis rotation (unitary) -> should commute with itself (of course) but not with others generally
Axis 1: sampling/reconstruction projector Π (idempotent-ish)
Axis 2: bandlimit projector in Laplacian eigenbasis (low modes)
You can increase m and supply more axes by following the same pattern.
"""
rng = np.random.default_rng(seed)
# Axis 0: orthonormal transform U
Q, _ = np.linalg.qr(rng.normal(size=(n, n)))
U = Q
def f0(x: np.ndarray, level: int) -> np.ndarray:
return U.T @ x
def g0(y: np.ndarray, level: int) -> np.ndarray:
return U @ y
ax0 = AxisMap(forward=f0, backward=g0, name="basis")
# Axis 1: projector Π = P P^T (rank r)
r = max(2, n // 3)
B = rng.normal(size=(n, r))
P = orth_project(B) # n x r with orthonormal cols
Pi = P @ P.T
def f1(x: np.ndarray, level: int) -> np.ndarray:
return Pi @ x
def g1(y: np.ndarray, level: int) -> np.ndarray:
# minimal-norm lift (here: same projector)
return Pi @ y
ax1 = AxisMap(forward=f1, backward=g1, name="project")
# Axis 2: bandlimit projector from a symmetric operator A (acts like "low-band corridor")
A = rng.normal(size=(n,n))
A = 0.5*(A + A.T)
keep = max(2, n // 4)
Pband = spectral_band_projector(A, keep=keep)
def f2(x: np.ndarray, level: int) -> np.ndarray:
# deeper levels tighten band (keep fewer)
k = max(1, keep - level)
Pb = spectral_band_projector(A, keep=k)
return Pb @ x
def g2(y: np.ndarray, level: int) -> np.ndarray:
# lift is same in this proxy
k = max(1, keep - level)
Pb = spectral_band_projector(A, keep=k)
return Pb @ y
ax2 = AxisMap(forward=f2, backward=g2, name="band")
return [ax0, ax1, ax2]

"""
